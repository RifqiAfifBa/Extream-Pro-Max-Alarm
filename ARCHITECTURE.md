```

     ↓

  provider.toggleAlarmActive(alarmId)
     ↓

     ↓

     ↓
Calls service.updateAlarm(updatedAlarm)
     ↓
Service updates SharedPreferences
     ↓
Provider notifies listeners
     ↓
UI rebuilds with updated toggle state ✅
```

### Timer Flow

```
User sets duration: 0h 2m 30s
     ↓
User taps Start
     ↓
TimerScreen:
  - Saves time to state variables
  - Sets _timerIsRunning = true
  - Calculates duration: 2min 30sec
     ↓
Starts Future.doWhile loop:
  - Waits 1 second
  - Calculates elapsed time
  - Updates remaining time
  - Updates UI (setState)
     ↓
Loop continues until:
  - remaining time = 0
  - OR user taps Pause
     ↓
When done:
  - Sets _timerIsRunning = false
  - Shows dialog notification
  - Plays sound/vibration ✅
```

## Key Components Explained

### 1. Models (Data Structures)

#### Alarm Model

```dart
class Alarm {
  // Unique identifier
  final String id;  // Generated by UUID

  // Time information
  final DateTime time;  // e.g., 2024-05-16 07:30:00

  // User information
  final String label;  // "Wake up", "Meeting", etc
  final bool isActive;  // Enabled/disabled toggle

  // Repeat settings
  final AlarmRepeat repeat;  // Never, Daily, Weekdays...
  final List<int> repeatDays;  // [1,3,5] = Mon, Wed, Fri

  // Behavior settings
  final String sound;  // "default", "gentle", etc
  final bool vibration;  // On/off
  final ChallengeType challengeType;  // QR, Math, Word, None

  // Metadata
  final DateTime createdAt;
  final DateTime? modifiedAt;
}
```

#### Timer Model

```dart
class Timer {
  final String id;
  final Duration duration;  // Total time to count down
  final String label;  // "Workout", "Cooking", etc
  final DateTime startTime;  // When timer started
  final bool isRunning;  // Active/paused
  final DateTime? pausedAt;  // When paused (if paused)

  // Calculates remaining time
  Duration getRemainingTime() {
    if (pausedAt != null) {
      // If paused, use saved pause time
      final elapsed = pausedAt!.difference(startTime);
      return duration - elapsed;
    } else if (isRunning) {
      // If running, calculate from now
      final elapsed = DateTime.now().difference(startTime);
      return duration - elapsed;
    }
    return duration;
  }
}
```

### 2. Services (Business Logic)

#### AlarmService - Key Methods

```dart
// Initialize (call once at app startup)
Future<void> init() async {
  _prefs = await SharedPreferences.getInstance();
}

// Get all alarms from storage
Future<List<Alarm>> getAlarms() async {
  final jsonString = _prefs.getString('alarms');
  if (jsonString == null) return [];

  final List<dynamic> jsonList = jsonDecode(jsonString);
  return jsonList.map((json) => Alarm.fromJson(json)).toList();
}

// Save alarms to storage
Future<void> saveAlarms(List<Alarm> alarms) async {
  final jsonList = alarms.map((alarm) => alarm.toJson()).toList();
  await _prefs.setString('alarms', jsonEncode(jsonList));
}

// Add new alarm
Future<void> addAlarm(Alarm alarm) async {
  final alarms = await getAlarms();
  alarms.add(alarm);
  await saveAlarms(alarms);
}

// Toggle alarm active status
Future<void> toggleAlarmActive(String alarmId) async {
  final alarms = await getAlarms();
  final index = alarms.indexWhere((a) => a.id == alarmId);
  if (index != -1) {
    final alarm = alarms[index];
    alarms[index] = alarm.copyWith(isActive: !alarm.isActive);
    await saveAlarms(alarms);
  }
}

// Find next alarm that will ring
Future<Alarm?> getNextAlarm() async {
  final alarms = await getAlarms();
  final activeAlarms = alarms.where((a) => a.isActive).toList();

  if (activeAlarms.isEmpty) return null;

  // Sort by time and return earliest
  activeAlarms.sort((a, b) => a.time.compareTo(b.time));
  return activeAlarms.first;
}
```

### 3. Provider (State Management)

#### AlarmProvider - Key Methods

```dart
class AlarmProvider extends ChangeNotifier {
  // State variables
  List<Alarm> _alarms = [];
  bool _isLoading = false;
  String? _error;

  // Getters
  List<Alarm> get alarms => _alarms;
  bool get isLoading => _isLoading;
  String? get error => _error;

  // Initialize
  Future<void> init() async {
    await _alarmService.init();
    await loadAlarms();
    // ...
  }

  // Load from service
  Future<void> loadAlarms() async {
    try {
      _alarms = await _alarmService.getAlarms();
      _alarms.sort((a, b) => a.time.compareTo(b.time));
      notifyListeners();  // Rebuild UI
    } catch (e) {
      _setError('Error: $e');
    }
  }

  // Add alarm
  Future<void> addAlarm(Alarm alarm) async {
    try {
      await _alarmService.addAlarm(alarm);
      await loadAlarms();  // Refresh
      _clearError();
    } catch (e) {
      _setError('Error: $e');
    }
  }

  // Notify listeners
  void _notifyUpdate() {
    notifyListeners();  // Triggers UI rebuild
  }
}
```

**How Provider Works:**

```
┌─ Provider State
│  ├─ _alarms: List<Alarm>
│  ├─ _isLoading: bool
│  └─ _error: String?
│
├─ Methods modify state
│  └─ Call notifyListeners()
│
└─ UI (Consumer) listens
   ├─ Rebuilds when notified
   └─ Shows updated data
```

### 4. UI Layer (Screens)

#### Using Consumer Pattern

```dart
// Consumer wraps widget to listen to provider
Consumer<AlarmProvider>(
  builder: (context, provider, _) {
    // This builder runs every time provider updates

    return ListView.builder(
      itemCount: provider.alarms.length,
      itemBuilder: (context, index) {
        final alarm = provider.alarms[index];

        return Container(
          // Show alarm data
          child: Text(alarm.label),
        );
      },
    );
  },
)
```

#### Update State from UI

```dart
// When user taps button
onTap: () {
  final provider = Provider.of<AlarmProvider>(context, listen: false);

  // Call provider method
  provider.toggleAlarmActive(alarmId);

  // Provider updates state → notifyListeners() → UI rebuilds
}
```

### Feature 3: Timer Countdown

```dart
void startTimer() {
  _timerStartTime = DateTime.now();
  _timerDuration = Duration(minutes: 5);
  _timerIsRunning = true;

  // Start countdown loop
  _startTimerTick();
}

void _startTimerTick() {
  Future.doWhile(() async {
    if (!_timerIsRunning) return false;

    // Wait 1 second
    await Future.delayed(Duration(seconds: 1));

    if (mounted) {
      // Calculate elapsed time
      final elapsed = DateTime.now().difference(_timerStartTime);
      final remaining = _timerDuration - elapsed;

      // Check if finished
      if (remaining.isNegative || remaining.inSeconds == 0) {
        _onTimerFinished();
        return false;
      }

      // Update UI
      setState(() {
        _timerSeconds = remaining.inSeconds % 60;
        _timerMinutes = remaining.inMinutes % 60;
        _timerHours = remaining.inHours;
      });
    }

    return _timerIsRunning;
  });
}

void _onTimerFinished() {
  setState(() => _timerIsRunning = false);

  // Show notification
  showDialog(
    context: context,
    builder: (context) => AlertDialog(
      title: Text('Timer Finished!'),
      actions: [
        TextButton(
          onPressed: () => Navigator.pop(context),
          child: Text('OK'),
        ),
      ],
    ),
  );
}
```

## Performance Considerations

### 1. **List Sorting**

- Alarms sorted by time for better UX
- Done in `loadAlarms()` to keep UI smooth

### 2. **LocalStorage Efficiency**

- Entire list saved as one JSON
- Faster than individual saves
- No database overhead

### 3. **UI Rebuilds**

- Provider only rebuilds affected widgets
- `Consumer` wraps only needed components
- Prevents unnecessary rebuilds

### 4. **Async Operations**

- All storage ops are async
- Doesn't block UI thread
- Smooth user experience
